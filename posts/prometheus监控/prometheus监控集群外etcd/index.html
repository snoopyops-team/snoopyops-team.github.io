通过其他主机访问 metrics 时，需要带上etcd证书，具体方式如下

1
2
3
4

ETCDCTL_CERT_FILE=/etc/ssl/certs/etcd-client.pem
ETCDCTL_KEY_FILE=/etc/ssl/certs/etcd-client-key.pem

curl -k --cert $ETCDCTL_CERT_FILE --key $ETCDCTL_KEY_FILE https://myHOST:2379/metrics

Prometheus监控

由于访问etcd metrics需要通过证书认证，先创建secret存放证书文件。

1
2
3
4

[root@ccp-prod-k8m1 ssl]# kubectl create secret generic etcd-client-cert --from-file=./ca.pem --from-file=./etcd.pem --from-file=./etcd-key.pem -n mon
[root@ccp-prod-k8m1 ssl]# kubectl get secrets -n mon etcd-client-cert
NAME               TYPE     DATA   AGE
etcd-client-cert   Opaque   3      17h

        volumeMounts:
        - mountPath: "/etc/prometheus"
          name: config-volume
        - mountPath: "/etc/alert-manager"
          name: config-alert
        - mountPath: "/prometheus"
          name: data
        # 将证书文件挂载到pod /etc/etcd-cert目录
        - mountPath: "/etc/etcd-cert"
          name: etcd-cert
          readOnly: true

      volumes:
      - name: data
        hostPath:
          path: /data/prometheus
          type: ""
      - name: config-volume
        configMap:
          name: prometheus-config
      - name: config-alert
        configMap:
          name: config-alert-prometheus
      # 挂载secret
      - name: etcd-cert
        secret:
          secretName: etcd-client-cert
          optional: true
